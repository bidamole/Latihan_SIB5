//Membuat Tabel

CREATE TABLE jenis( 
    id int PRIMARY KEY AUTO_INCREMENT, 
    nama varchar (30) not null unique);

CREATE TABLE produk (
    id int PRIMARY KEY AUTO_INCREMENT,
    nama varchar(30) NOT NULL UNIQUE,
    harga double NOT NULL,
    stok int NOT NULL,
    idjenis int NOT NULL REFERENCES jenis(id),
    foto varchar(30));

1. Id => adalah nama kolom atau header kolom.
2. Int => adalah tipe data angka.
3. PRIMARY KEY => adalah kunci dari sebuah tabel.
4. AUTO_INCREMENT => adalah key untuk otomatisasi penomoran.
5. NOT NULL => Menandakkan kolom tersebut tidak boleh kosong jika data mulai diisi.
6. UNIQUE => adalah key untuk kolom agar data tidak boleh sama.
7. YES NULL => Boleh dikosongkan.
8. REFERENCES => Merujuk pada tabel tertentu atau saling keterkaitan.
